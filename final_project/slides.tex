% 4. IMPACT-STATE RECURSION
% ============================================================
\begin{frame}{Impact-state recursion \hfill \emph{\S2.2}}{OW vs AFS -- different flow kernels, same decay}

  \begin{columns}[T]
  \column{0.50\textwidth}
  \footnotesize
  Both models build the OU recursion
  \[
    \bar I_{t+1} = (1-\beta)\bar I_t + \tilde q_t,
    \qquad \beta = \frac{\ln 2}{H_{\text{bins}}}.
  \]
  They differ only in the \hilite{flow kernel} -- how a child trade of size $q_t$ pushes the state $\bar I$:
  \[
    \text{\hilite{OW}:}\;\;\tilde q_t = \sigma\,q_t/\mathrm{ADV}\;\;(\text{linear})
  \]
  \[
    \text{\hilite{AFS}:}\;\;\tilde q_t = \sigma\,\mathrm{sgn}(q_t)\sqrt{|q_t|/\mathrm{ADV}}\;\;(\text{concave})
  \]

  \vspace{0.4em}
  \textbf{Why AFS is interesting.}
  The model lets us \emph{specify} the flow kernel $f(q)$ rather than inheriting the linear form by default. We pick the sqrt because it lines up with the empirical $\sqrt{Q/V}$ meta-order law -- mechanism explored on \emph{slide 9} (Backtest engine), where the kernel choice becomes the operative modelling decision.

  \column{0.48\textwidth}
  \footnotesize
  \textbf{Optimisation problem.} Joint
  \[
    \max_{H,\,\alpha,\,\lambda}\; R^2_{\text{OOS}}\!\bigl(H,\alpha,\lambda\bigr).
  \]
  For fixed $H$, OLS pins $(\hat\alpha,\hat\lambda)$ in closed form, so the problem reduces to a \hilite{1-D scan over $H$} -- the half-life that yields the best-fitting impact regression:
  \[
    \max_{H}\; R^2_{\text{OOS}}\!\bigl(H,\,\hat\alpha(H),\,\hat\lambda(H)\bigr).
  \]

  \vspace{0.4em}
  \textbf{Regression target.} At horizon $\tau = 6$ bins (60\,s),
  \[
    y_t = \alpha + \lambda\,x_t + \varepsilon_t,
  \]
  with $x_t = \bar I_t - \bar I_{t-\tau}$, $y_t = (P_t - P_{t-\tau})/P_{t-\tau}$. Fit via additive sufficient statistics $(S_x, S_y, S_{xx}, S_{yy}, S_{xy}, n)$ -- enables fast rolling re-fits and the $H$-grid sweep on the next slide.
  \end{columns}
\end{frame}

\begin{frame}{Non-parametric extension \hfill \warn{ENHANCEMENT \#1}}{Method + estimated functional form}

  \begin{columns}[T]
  \column{0.40\textwidth}
  \footnotesize
  Replace the parametric $y = \alpha + \lambda x$ with a binned $g(x)$, applied to \emph{both} OW and AFS (same procedure, different $x_t = \Delta\bar I$):
  \[
    g_{i,b}^{\mathrm{reg}}(\gamma)
    \;=\; \frac{S_{y,i,b} + \gamma\,\bar g_b}{n_{i,b}+\gamma}.
  \]
  \begin{itemize}
    \item $B = 15$ quantile bins on train month $m$ (pooled across stocks)
    \item $\bar g_b$ = pooled universal curve across all stocks
    \item Per-stock $\hat g_{i,b}$ shrunk toward $\bar g_b$ with strength $\gamma$
    \item $\gamma$ tuned on test month $m{+}1$, validated on $m{+}2$
  \end{itemize}

  \centering

\begin{frame}{Backtest engine \hfill \warn{\S2.3 -- ENHANCEMENT \#2}}{Waelbroeck + AFS sqrt as the explicit modelling choice + carry modes}

  \[
    P_{\text{sim}}(t) = P_0\bigl(1 + \mathrm{cumret}(t) + g_{\text{sim}}(t) - g_{\text{ref}}(t)\bigr),
    \quad g(t) = \lambda\,\bar I(t)
  \]
  $\bar I^{\text{ref}}$ uses public-tape \texttt{orderFlow}; $\bar I^{\text{sim}}$ uses the engine's input trade path.

  \begin{columns}[T]
  \column{0.52\textwidth}
  \footnotesize
  \textbf{Modelling choice: why AFS sqrt?} A constant-rate parent of size $Q$ over $T$ drives the OU recursion to equilibrium $\bar I^* = f(Q/T)/\beta$, so
  \[
    \frac{\Delta P}{P_0} \;\propto\; f(Q/T).
  \]
  The kernel's shape \emph{propagates directly} to the parent (decay is linear -- $1/\beta$ rescale, not reshape):
  \begin{itemize}
    \item \warn{OW} ($f = q$): $\Delta P \propto Q/V$ -- linear in size
    \item \good{AFS} ($f = \sqrt{q}$): $\Delta P \propto \sqrt{Q/V}$ -- the empirical \hilite{Almgren / Bouchaud law}
  \end{itemize}
  Picking AFS at the engine level wires the empirical regularity in by construction.

  \column{0.46\textwidth}
  \footnotesize
  \textbf{Generic trades.} Engine accepts any signed schedule via \texttt{trade\_provider}: \texttt{make\_optimal\_provider} (OW/AFS) and \texttt{make\_fixed\_provider} (TWAP/VWAP/manual).

  \vspace{0.3em}
  \textbf{Carry modes.}
  \begin{itemize}
    \item \texttt{daily} -- reset $\bar I$ + inventory each session
    \item \texttt{multi} -- persist with overnight decay $(1-\beta)^{16\text{h}\cdot 6}$ on $\bar I$; \hilite{inventory carries regardless}
  \end{itemize}

  \vspace{0.3em}
  \textbf{Stock splits.} Heuristic detector flags $|\Delta p|$, $|\Delta v|$ opposite-signed + stable notional. 2019 top-20: \warn{none detected}.
  \end{columns}
\end{frame}

% 9. SYNTHETIC ALPHA + OPTIMAL STRATEGY

\begin{frame}{Synthetic $\alpha$ + optimal strategy \hfill \emph{\S2.4--\S2.5}}{Formulas; OW vs AFS use the same target-position rule}

  \begin{columns}[T]
  \column{0.5\textwidth}
  \footnotesize
  \textbf{Synthetic alpha} (\S2.4)
  \[
    \alpha_t = r^h_t + y\,\frac{dW_t}{P_t},\quad
    y = \sqrt{(\rho^{-2}{-}1)\frac{\mathrm{Var}(r^h)}{E[P^{-2}]h}}
  \]
  Choosing $x{=}1$ in $\alpha = x\,r^h + y\,dW/P$ gives:
  \begin{itemize}
    \item \hilite{Unbiased}: $E[r^h\mid\alpha_t] = \alpha_t$
    \item \hilite{Target correlation}: $\mathrm{Corr}(\alpha, r^h) = \rho$
  \end{itemize}
  We use $\rho = 0.05$, $h=1$ bin (10\,s); empirical $\rho$ within $\pm 1\%$ of target.

  \column{0.5\textwidth}
  \footnotesize
  \textbf{OW closed form} (\S2.5)
  \[
    X^*_t = \frac{\alpha_t\,\mathrm{ADV}}{2\lambda\sigma}\;,\quad
    \kappa = \beta = \frac{\ln 2}{H_{\text{bins}}}
  \]
  Position adjusts to $X^*$ at speed $\kappa$; ramped to zero over the final 30 min.

  \vspace{0.3em}
  \textbf{AFS variant.}
  Same target rule per \texttt{project.ipynb}; sqrt nonlinearity enters via the simulator's $\tilde q$.

  \vspace{0.3em}
  \textbf{Ext-OW with $\lambda(t)$.}
  \warn{Placeholder.} Simulator already accepts a per-bin $\lambda_t$ array; HJB-optimal trade rule is not $\kappa$-constant when $\lambda$ varies -- closed form pending.
  \end{columns}

  \vspace{0.3em}
  \footnotesize\itshape
  Position trajectory sketch: $X_t$ rises toward $X^*_t$ at exponential speed $\kappa$, clips at $\pm 0.5\%\,\mathrm{ADV}$, ramps linearly to zero in the last 30 min. OW and AFS positions coincide; the cost accounting differs.
\end{frame}

\begin{frame}{What we learned}{Re-stating the two enhancements with one bullet each}

  \begin{block}{Enhancement 1 -- Non-parametric impact with $\gamma$ regularisation \hfill (\S2.2)}
    \footnotesize
    Apples-to-apples pooled OOS $R^2$: \hilite{OW $0.161 \to 0.221$} ($+5.9$\,pp, linearity binding);\;
    \hilite{AFS unchanged at $0.263$} (square-root on the efficient frontier; NP buys nothing).\;
    $\gamma$ shrinkage immaterial in the data-rich headline regime; \hilite{$\gamma^*/n$ rises $6\times$ and OOS gain $14\times$} as the training window shrinks 10$\times$ -- the estimator behaves correctly under stress.
  \end{block}

  \begin{block}{Enhancement 2 -- Multi-day carry backtest \hfill (\S2.3)}
    \footnotesize
    Carrying $\bar I$ and inventory across sessions \hilite{quadruples gross P\&L} ($\$118$k $\to$ $\$428$k) -- accumulated inventory rides the trend. Daily-reset wins on Sharpe (+3.43 vs +2.05) with tight drawdowns; multi-day wins on absolute dollars but introduces a subtle artefact: under Waelbroeck mark-to-market the carried long position benefits from its own simulated price displacement, producing a \warn{negative reported impact cost} ($-\$39$k). This is an accounting effect, not free money -- a clean impact-cost decomposition for multi-day inventory needs a different framing (e.g.\ implementation-shortfall against an arrival-price benchmark). The engine handles both modes via a single flag.
  \end{block}
